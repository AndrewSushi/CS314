5b. 0x3FC00000 = 1.5
	3    F    C    0    0    0    0    0
	0011 1111 1100 0000 0000 0000 0000 0000
	sign = 0
	exp = 01111111
	mant = 1000...
	restore mantisa leading 1
	1.1000...
	k = 8, 2^(8-1) - 1 = 2^7 - 1 = 127 = bias
	exp - bias = 0b01111111 - 127 = 127 - 127 = 0
	1.1000... * 2^(0) = 1.1
	2^(0) 2^(-1)
	1     0.5
	sum: 1 + 0.5 = 1.5
